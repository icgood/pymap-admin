    def handle_response(self, response: HealthCheckResponse,
                        outfile: TextIO, errfile: TextIO) -> int:
        success = self._is_serving(response)
        if success:
            print(response, file=outfile)
            return 0
        else:
            print(response, file=errfile)
            return 1
